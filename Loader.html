    <body>
      <div class="fan-container">
        <div class="fan" id="fan"></div>
      </div>
      <button class="control-button" id="control-button">Turn On</button>
      <script>
        const fan = document.getElementById('fan');
        const controlButton = document.getElementById('control-button');
        let isFanOn = false;
        let fanRotation = 0;
    
        controlButton.addEventListener('click', () => {
          isFanOn = !isFanOn;
    
          if (isFanOn) {
            fan.style.animation = `spin 1s linear infinite`;
            controlButton.textContent = 'Turn Off';
          } else {
            fanRotation = getCurrentRotation();
            fan.style.animation = 'none'; // Stop animation
            fan.style.transform = `rotate(${fanRotation}deg)`; // Set fan to stop at current rotation
            controlButton.textContent = 'Turn On';
          }
        });
    
        function getCurrentRotation() {
          const style = window.getComputedStyle(fan);
          const transform = style.getPropertyValue('transform');
          const matrix = transform.match(/^matrix\((.+)\)$/);
          if (matrix) {
            const values = matrix[1].split(',');
            return Math.round(Math.atan2(values[1], values[0]) * (180 / Math.PI));
          }
          return 0;
        }
      </script>
