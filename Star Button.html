<script>
  document.querySelector('.breaking-button').addEventListener('click', function () {
    const button = this;
    const buttonRect = button.getBoundingClientRect();
    for (let i = 0; i < 50; i++) { // Increased the number of stars to 50
      const star = document.createElement('div');
      star.classList.add('star');
      star.style.left = `${Math.random() * buttonRect.width}px`;
      star.style.top = `${Math.random() * buttonRect.height}px`;
      button.appendChild(star);

      setTimeout(() => {
        star.style.transform = `translate(${Math.random() * 300 - 150}px, ${Math.random() * 300 - 150}px) rotate(${Math.random() * 1080}deg)`;
        star.style.opacity = '1';
      }, 100);

      star.addEventListener('transitionend', () => {
        star.remove();
      });
    }
  });
</script>
