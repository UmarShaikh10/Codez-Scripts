          <div class="battery-fill" id="battery-fill"></div>
        </div>
      
        <script>
          let fillLevel = 0;
          let increasing = true;
      
          function animateBattery() {
            const batteryFill = document.getElementById('battery-fill');
      
            if (increasing) {
              fillLevel += 10;
              if (fillLevel >= 100) {
                fillLevel = 100;
                increasing = false;
              }
            } else {
              fillLevel -= 10;
              if (fillLevel <= 0) {
                fillLevel = 0;
                increasing = true;
              }
            }
      
            batteryFill.style.width = `${fillLevel}%`;
      
            // Calculate color based on fill level
            let green = Math.round((fillLevel / 100) * 255);
            let red = 255 - green;
            let color = `rgb(${red}, ${green}, 0)`;
            batteryFill.style.backgroundColor = color;
      
            // Set box-shadow to create a glowing effect
            batteryFill.style.boxShadow = `0 0 20px ${color}`;
          }
      
          // Animate the battery every half second
          setInterval(animateBattery, 500);
      
        </script>
      </body>
